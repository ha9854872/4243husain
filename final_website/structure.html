    • 1. Employee structure with ID, name, designation for 50 employees in tabular form. 
    //create structure of employee which there is emp.id and so on
#include <stdio.h>
struct Employee
{
int id;
char name[50];
char designation[50];
};
int main()
{
struct Employee emp[50];
int i, n;
printf("Enter number of employees (max 50): ");
scanf("%d", &n);
for(i = 0; i < n; i++)
{
printf("\nEnter details of employee %d:\n", i + 1);
printf("Enter Employee ID: ");
scanf("%d", &emp[i].id);
printf("Enter Name: ");
scanf("%s", emp[i].name);
printf("Enter Designation: ");
scanf("%s", emp[i].designation);
}
printf("\n---------------------------------------------------------\n");
printf("Emp.ID\t\tName\t\tDesignation\n");
printf("---------------------------------------------------------\n");
for(i = 0; i < n; i++)
{
printf("%d\t\t%-10s\t%-10s\n", emp[i].id, emp[i].name, emp[i].designation);
}
printf("---------------------------------------------------------\n");
}

    4. Student structure: roll name, fee, DOB, print in ascending order by roll. 
    #include <stdio.h>
struct Student
{
int roll;
char name[50];
float fee;
char dob[15];  
};
int main()
{
struct Student s[50], temp;
int i, j, n;
printf("Enter number of students (max 50): ");
scanf("%d", &n);
for(i = 0; i < n; i++)
{
printf("\nEnter details of student %d:\n", i + 1);
printf("Enter Roll No: ");
scanf("%d", &s[i].roll);
printf("Enter Name: ");
scanf("%s", s[i].name);
printf("Enter Fee: ");
scanf("%f", &s[i].fee);
printf("Enter DOB (dd-mm-yyyy): ");
scanf("%s", s[i].dob);
}
// Sorting by roll number (ascending)
for(i = 0; i < n - 1; i++)
{
for(j = i + 1; j < n; j++)
{
if(s[i].roll > s[j].roll)
{
temp = s[i];
s[i] = s[j];
s[j] = temp;
}
}
}
printf("\n-----------------------------------------------\n");
printf("Roll\tName\t\tFee\t\tDOB\n");
printf("-----------------------------------------------\n");
for(i = 0; i < n; i++)
{
printf("%d\t%-10s\t%.2f\t\t%s\n", s[i].roll, s[i].name, s[i].fee, s[i].dob);
}
printf("-----------------------------------------------\n");

}
