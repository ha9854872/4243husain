          5. Swap numbers using Call by reference.
         
     #include <stdio.h>
void swap(int *a, int *b) {
    int temp;

    temp = *a;   // store value of a in temp
    *a = *b;     // put value of b in a
    *b = temp;   // put value of temp (old a) in b

    printf("Inside function after swapping: a = %d, b = %d\n", *a, *b);
}

void main() {
    int x, y;

    printf("Enter first number: ");
    scanf("%d", &x);

    printf("Enter second number: ");
    scanf("%d", &y);

    printf("\nBefore swapping: x = %d, y = %d\n", x, y);

    swap(&x, &y);   // call function and pass addresses

    printf("After swapping (in main): x = %d, y = %d\n", x, y);
}
